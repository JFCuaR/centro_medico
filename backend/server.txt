const verificarToken = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) return res.sendStatus(401);

  jwt.verify(token, 'secreto', (err, user) => {
    if (err) return res.sendStatus(403);
    req.user = user;
    next();
  });
};

async function getOrCreateCajaAbierta(db, idUsuario) {
  // ¿Tiene caja abierta?
  const [abierta] = await db.query(
    "SELECT * FROM caja WHERE id_usuario = ? AND estado = 'abierta' LIMIT 1",
    [idUsuario]
  );
  if (abierta.length) return abierta[0];

  // Si no, crear una nueva
  const [ins] = await db.query(
    "INSERT INTO caja (id_usuario, estado) VALUES (?, 'abierta')",
    [idUsuario]
  );
  const [nueva] = await db.query(
    "SELECT * FROM caja WHERE id_caja = ?",
    [ins.insertId]
  );
  return nueva[0];
}

async function calcularTotalCaja(db, idCaja) {
  const [tot] = await db.query(
    `SELECT COALESCE(SUM(dv.cantidad * dv.precio_unitario),0) AS total
     FROM ventas v
     JOIN detalles_venta dv ON dv.id_venta = v.id_venta
     WHERE v.id_caja = ? AND (dv.devuelto = 0 OR dv.devuelto IS NULL)`,
    [idCaja]
  );
  return Number(tot[0].total || 0);
}

// Reemplazo completo de /realizarventa
app.post('/realizarventa', async (req, res) => {
  const cart = req.body;
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  let connection; // ✅ <-- declarar antes del try

  if (!token) return res.status(401).json({ message: 'Token no proporcionado' });

  try {
    const decoded = jwt.verify(token, 'secreto');
    const idUsuario = decoded.id;

    // 🔹 Buscar la caja abierta del usuario
    const [cajas] = await db.query(
      "SELECT id_caja FROM caja WHERE id_usuario = ? AND estado = 'abierta' LIMIT 1",
      [idUsuario]
    );
    if (!cajas.length) {
      return res.status(400).json({ message: 'No hay caja abierta para este usuario' });
    }
    const idCaja = cajas[0].id_caja;

    console.log(`🧾 Venta vinculada a usuario ${idUsuario} y caja ${idCaja}`);

    // ✅ ahora sí asignamos la conexión
    connection = await db.getConnection();
    await connection.beginTransaction();

    // Crear la venta vinculada al usuario y la caja
    const [ventaResult] = await connection.query(
      'INSERT INTO ventas (fecha, id_usuario, id_caja) VALUES (NOW(), ?, ?)',
      [idUsuario, idCaja]
    );
    const idVenta = ventaResult.insertId;

    // Procesar cada ítem del carrito
    for (let item of cart) {
      const id_lote = item.id_lote;
      const cantidad = item.cantidad || 0;

      if (isNaN(cantidad) || cantidad <= 0) {
        throw new Error('Cantidad inválida');
      }

      const [rows] = await connection.query(
        `SELECT lotes.*, medicamentos.nombre
         FROM lotes
         JOIN medicamentos ON lotes.id_medicamento = medicamentos.id_medicamento
         WHERE lotes.id_lote = ?`,
        [id_lote]
      );

      if (rows.length === 0) throw new Error(`Lote ${id_lote} no encontrado`);

      const lote = rows[0];
      if (lote.stock < cantidad) throw new Error(`Stock insuficiente para ${lote.nombre}`);

      await connection.query(
        `UPDATE lotes SET stock = stock - ? WHERE id_lote = ?`,
        [cantidad, id_lote]
      );

      await connection.query(
        `INSERT INTO detalles_venta
         (id_venta, id_medicamento, cantidad, precio_unitario, id_lote)
         VALUES (?, ?, ?, ?, ?)`,
        [idVenta, lote.id_medicamento, cantidad, lote.precio_venta, id_lote]
      );
    }

    await connection.commit();
    res.status(200).json({ message: 'Venta realizada con éxito', idVenta, idCaja });

  } catch (err) {
    console.error('❌ Error al realizar la venta:', err);
    if (connection) await connection.rollback(); // ahora sí existe la variable
    res.status(500).json({ message: 'Error al realizar la venta', error: err.message });
  } finally {
    if (connection) connection.release(); // ✅ ya no rompe
  }
});

app.post('/caja/cerrar', verificarToken, async (req, res) => {
  const idUsuario = req.user.id;
  let conn;
  try {
    // Buscar caja abierta
    const [abiertas] = await db.query(
      "SELECT * FROM caja WHERE id_usuario = ? AND estado = 'abierta' LIMIT 1",
      [idUsuario]
    );
    if (!abiertas.length) {
      return res.status(400).json({ msg: 'No tenés caja abierta.' });
    }
    const caja = abiertas[0];

    const total = await calcularTotalCaja(db, caja.id_caja);

    conn = await db.getConnection();
    await conn.beginTransaction();
    await conn.query(
      `UPDATE caja
       SET estado='cerrada', fecha_cierre=NOW(), total_ventas=?
       WHERE id_caja = ?`,
      [total, caja.id_caja]
    );
    await conn.commit();

    res.json({
      msg: 'Caja cerrada correctamente',
      idCaja: caja.id_caja,
      total: total.toFixed(2)
    });
  } catch (e) {
    if (conn) await conn.rollback();
    console.error(e);
    res.status(500).json({ error: 'No se pudo cerrar la caja' });
  } finally {
    if (conn) conn.release();
  }
});

app.post('/historial-medico', async (req, res) => {
  const {
    nombre_paciente,
    fecha_consulta,
    diagnostico,
    sintomas,
    sexo,
    religion,
    medico_responsable,
    medicamentos_recetados,
    dpi,
    telefono,
    fecha_nacimiento,
    antecedentes_medico,
    antecedentes_quirurgico,
    antecedentes_alergico,
    antecedentes_traumaticos,
    antecedentes_familiares,
    antecedentes_vicios_y_manias
  } = req.body;

  try {
    // 1. Verificar si el paciente ya existe
    const [pacienteRows] = await db.query('SELECT id_paciente FROM pacientes WHERE dpi = ?', [dpi]);
    let id_paciente;

    if (pacienteRows.length > 0) {
      id_paciente = pacienteRows[0].id_paciente;
    } else {
      const [insertPaciente] = await db.query(`
          INSERT INTO pacientes (nombre, dpi, telefono, fecha_nacimiento, sexo, religion)
          VALUES (?, ?, ?, ?, ?, ?)`,
        [nombre_paciente, dpi, telefono, fecha_nacimiento, sexo, religion]
      );
      id_paciente = insertPaciente.insertId;
    }

    // 2. Formatear los medicamentos recetados
    const medicamentosFormateados = medicamentos_recetados
      .filter(m => m.nombre && typeof m.nombre === 'string')
      .map(m => {
        const modo = m.modo_administracion || '';
        const cantidad = m.cantidad || '';
        const unidad = m.unidad || '';
        const comentario = m.comentario ? ` ${m.comentario}` : '';
        return `${m.nombre} ${modo} ${cantidad} ${unidad}${comentario}`.trim();
      })
      .join(' | ');

    // 3. Insertar en historial_medico
    const [historialInsert] = await db.query(
      `INSERT INTO historial_medico 
        (id_paciente, fecha_consulta, diagnostico, sintomas, medico_responsable, medicamentos_recetados)
        VALUES (?, ?, ?, ?, ?, ?)`,
      [id_paciente, fecha_consulta, diagnostico, sintomas, medico_responsable, medicamentosFormateados]
    );

    const id_historial = historialInsert.insertId;

    // 4. Insertar antecedentes médicos
    await db.query(
      `INSERT INTO antecedentes_medicos 
        (id_historial, antecedentes_medico, antecedentes_quirurgico, antecedentes_alergico, antecedentes_traumaticos, antecedentes_familiares, antecedentes_vicios_y_manias) 
        VALUES (?, ?, ?, ?, ?, ?, ?)`,
      [
        id_historial,
        antecedentes_medico,
        antecedentes_quirurgico,
        antecedentes_alergico,
        antecedentes_traumaticos,
        antecedentes_familiares,
        antecedentes_vicios_y_manias,
      ]
    );

    res.status(201).send('Historial médico y antecedentes guardados exitosamente');

  } catch (err) {
    console.error('Error al guardar el historial médico y antecedentes:', err);
    res.status(500).send('Error al guardar el historial médico y antecedentes');
  }
});

app.get('/sales/:date', async (req, res) => {
  const selectedDate = req.params.date;

  const queryAllSales = `
    SELECT 
      dv.id_medicamento, 
      SUM(dv.cantidad) AS cantidad_total, 
      dv.precio_unitario, 
      m.nombre AS medicamento, 
      dv.devuelto, 
      SUM(dv.cantidad * dv.precio_unitario) AS total_ventas
    FROM detalles_venta dv
    JOIN lotes l ON dv.id_lote = l.id_lote
    JOIN medicamentos m ON l.id_medicamento = m.id_medicamento
    JOIN ventas v ON dv.id_venta = v.id_venta
    WHERE DATE(v.fecha) = ? 
    GROUP BY dv.id_medicamento, dv.precio_unitario, m.nombre, dv.devuelto;
  `;

  const queryNonReturnedSales = `
    SELECT 
      SUM(dv.cantidad * dv.precio_unitario) AS total_ventas,
      SUM(dv.cantidad) AS cantidad_total, 
      dv.precio_unitario, 
      SUM(dv.cantidad * l.precio_compra) AS total_cost
    FROM detalles_venta dv
    JOIN lotes l ON dv.id_lote = l.id_lote
    JOIN medicamentos m ON l.id_medicamento = m.id_medicamento
    JOIN ventas v ON dv.id_venta = v.id_venta
    WHERE DATE(v.fecha) = ? AND dv.devuelto = 0
    GROUP BY dv.id_medicamento, dv.precio_unitario;
  `;

  try {
    const [allSalesResults] = await db.query(queryAllSales, [selectedDate]);
    const [nonReturnedResults] = await db.query(queryNonReturnedSales, [selectedDate]);

    let totalSales = 0;
    let totalCost = 0;

    nonReturnedResults.forEach(item => {
      totalSales += parseFloat(item.total_ventas) || 0;
      totalCost += parseFloat(item.total_cost) || 0;
    });

    res.json({
      totalSales: totalSales.toFixed(2),
      totalCost: totalCost.toFixed(2),
      profit: (totalSales - totalCost).toFixed(2),
      details: allSalesResults.map(item => ({
        medicamento: item.medicamento,
        cantidad: item.cantidad_total,
        precio_unitario: item.precio_unitario,
        devuelto: item.devuelto
      }))
    });

// 📆 REPORTE SEMANAL
app.get('/sales/week/:date', async (req, res) => {
  const selectedDate = req.params.date;

  const queryAllSales = `
    SELECT 
      dv.id_medicamento, 
      SUM(dv.cantidad) AS cantidad_total, 
      dv.precio_unitario, 
      m.nombre AS medicamento, 
      dv.devuelto, 
      SUM(dv.cantidad * dv.precio_unitario) AS total_ventas
    FROM detalles_venta dv
    JOIN lotes l ON dv.id_lote = l.id_lote
    JOIN medicamentos m ON l.id_medicamento = m.id_medicamento
    JOIN ventas v ON dv.id_venta = v.id_venta
    WHERE WEEK(v.fecha) = WEEK(?)
    GROUP BY dv.id_medicamento, dv.precio_unitario, m.nombre, dv.devuelto;
  `;

  const queryNonReturnedSales = `
    SELECT 
      SUM(dv.cantidad * dv.precio_unitario) AS total_ventas,
      SUM(dv.cantidad) AS cantidad_total, 
      dv.precio_unitario, 
      SUM(dv.cantidad * l.precio_compra) AS total_cost
    FROM detalles_venta dv
    JOIN lotes l ON dv.id_lote = l.id_lote
    JOIN medicamentos m ON l.id_medicamento = m.id_medicamento
    JOIN ventas v ON dv.id_venta = v.id_venta
    WHERE WEEK(v.fecha) = WEEK(?) AND dv.devuelto = 0
    GROUP BY dv.id_medicamento, dv.precio_unitario;
  `;

  try {
    const [allSalesResults] = await db.query(queryAllSales, [selectedDate]);
    const [nonReturnedResults] = await db.query(queryNonReturnedSales, [selectedDate]);

    let totalSales = 0;
    let totalCost = 0;

    nonReturnedResults.forEach(item => {
      totalSales += parseFloat(item.total_ventas) || 0;
      totalCost += parseFloat(item.total_cost) || 0;
    });

    res.json({
      totalSales: totalSales.toFixed(2),
      totalCost: totalCost.toFixed(2),
      profit: (totalSales - totalCost).toFixed(2),
      details: allSalesResults.map(item => ({
        medicamento: item.medicamento,
        cantidad: item.cantidad_total,
        precio_unitario: item.precio_unitario,
        devuelto: item.devuelto
      }))
    });
  } catch (err) {
    console.error('Error al obtener el reporte semanal:', err);
    res.status(500).json({ error: 'Error al obtener el reporte semanal' });
  }
});

app.post('/devolver_producto', async (req, res) => {
  const { id_venta, id_medicamento, cantidad_devuelta } = req.body;

  try {
    // Buscar el producto en el detalle de la venta
    const [detalleVenta] = await db.query(
      'SELECT id_lote, cantidad, devuelto FROM detalles_venta WHERE id_venta = ? AND id_medicamento = ?',
      [id_venta, id_medicamento]
    );

    if (detalleVenta.length === 0) {
      return res.status(404).json({ error: 'Detalle de venta no encontrado' });
    }

    const { id_lote, cantidad, devuelto } = detalleVenta[0];

    if (devuelto) {
      return res.status(400).json({ error: 'El producto ya fue devuelto' });
    }

    if (cantidad_devuelta > cantidad) {
      return res.status(400).json({ error: 'Cantidad devuelta excede la cantidad vendida' });
    }

    // Actualizar el stock en el lote
    await db.query(
      'UPDATE lotes SET stock = stock + ? WHERE id_lote = ?',
      [cantidad_devuelta, id_lote]
    );

    // Marcar como devuelto
    await db.query(
      'UPDATE detalles_venta SET devuelto = TRUE WHERE id_venta = ? AND id_medicamento = ?',
      [id_venta, id_medicamento]
    );

    res.status(200).json({ message: 'Producto devuelto y stock actualizado' });
  } catch (error) {
    console.error('Error en la devolución:', error);
    res.status(500).json({ error: 'Error en la devolución', details: error.message });
  }
});

// ✅ MODIFICADO: POST registrar encamamiento usando id_paciente
app.post('/api/encamamientos', async (req, res) => {
  const { numero, id_paciente, estado, dieta, ingreso, observaciones } = req.body;

  try {
    if (!id_paciente) {
      return res.status(400).json({ error: 'id_paciente es requerido' });
    }

    // Verificar si el cuarto ya está ocupado
    const [ocupado] = await db.query(`
      SELECT * FROM encamamiento 
      WHERE no_cuarto = ? AND fecha_salida IS NULL
      LIMIT 1
    `, [numero]);

    if (ocupado.length > 0) {
      return res.status(400).json({ error: `El cuarto ${numero} ya está ocupado por otro paciente.` });
    }

    // Obtener el id_historial más reciente del paciente
    const [historial] = await db.query(`
      SELECT id
      FROM historial_medico 
      WHERE id_paciente = ?
      ORDER BY fecha_consulta DESC
      LIMIT 1
    `, [id_paciente]);

    const id_historial = historial.length > 0 ? historial[0].id : null;

    // Insertar encamamiento con id_historial
    const [enc] = await db.query(`
      INSERT INTO encamamiento (id_paciente, no_cuarto, fecha_ingreso, ocupado, id_historial)
      VALUES (?, ?, ?, 1, ?)
    `, [id_paciente, numero, ingreso, id_historial]);

    const id_encamamiento = enc.insertId;

    // Insertar en paciente_interno
    await db.query(`
      INSERT INTO paciente_interno (id_encamamiento, estado, dieta, observaciones)
      VALUES (?, ?, ?, ?)
    `, [id_encamamiento, estado, dieta, observaciones]);

    // Devolver cuartos ocupados actualizados (incluyendo id_historial)
    const [rows] = await db.query(`
      SELECT 
        e.id_encamamiento,
        CAST(e.no_cuarto AS UNSIGNED) AS numero,
        p.nombre AS paciente,
        pi.estado,
        pi.dieta,
        e.fecha_ingreso AS ingreso,
        pi.observaciones,
        e.fecha_salida,
        e.observacion_salida AS observacion_salida,
        e.ocupado,
        e.id_historial
      FROM encamamiento e
      LEFT JOIN paciente_interno pi ON pi.id_encamamiento = e.id_encamamiento
      LEFT JOIN pacientes p ON p.id_paciente = e.id_paciente
      WHERE e.ocupado = 1
      ORDER BY e.no_cuarto
    `);

    res.json({ cuartos: rows });
  } catch (err) {
    console.error('Error al registrar encamamiento:', err);
    res.status(500).json({ error: 'Error al registrar encamamiento' });
  }
});

// pasa un turno a "atendiendo" y quita cualquier otro activo
app.put('/api/turnos/:id/atender', async (req, res) => {
  const { id } = req.params;
  const conn = await db.getConnection();
  try {
    await conn.beginTransaction();
    await conn.execute('UPDATE turnos SET estado="espera" WHERE estado="atendiendo"'); // deja solo 1 activo
    await conn.execute('UPDATE turnos SET estado="atendiendo" WHERE id_turno=?', [id]);
    await conn.commit();
    res.sendStatus(200);
  } catch (e) {
    await conn.rollback();
    console.error(e);
    res.status(500).json({ error: 'No se pudo poner en atendiendo' });
  } finally {
    conn.release();
  }
});

app.get('/buscar_antecedentes', async (req, res) => {
  const { tipo, termino } = req.query;

  if (!['medico', 'quirurgico', 'alergico', 'traumaticos', 'familiares', 'vicios_y_manias'].includes(tipo)) {
    return res.status(400).json({ error: 'Tipo de antecedente no válido' });
  }

  const columna = `antecedentes_${tipo}`;

  try {
    const [resultados] = await db.query(
      `SELECT DISTINCT ${columna} FROM antecedentes_medicos WHERE ${columna} LIKE ? LIMIT 5`,
      [`%${termino}%`]
    );

    const sugerencias = resultados.map((fila) => fila[columna]);
    res.json(sugerencias);
  } catch (error) {
    console.error('Error al buscar antecedentes:', error);
    res.status(500).json({ error: 'Error al buscar antecedentes' });
  }
});
